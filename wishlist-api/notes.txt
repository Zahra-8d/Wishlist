Tables
----------

1. Lists 
    id
    owner_id (user id)
    name
    description
    date

CREATE TABLE lists(
id INT AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(125),
description VARCHAR(125),
owner_id INT(10),
date DATE
);

3. Gifts
    id
    list_id
    name
    description
    url
    checked (bool)
    checked_by (user id)
    checked_date

CREATE TABLE gifts(
id INT AUTO_INCREMENT PRIMARY KEY,
list_id INT(10),
name VARCHAR(125),
description VARCHAR(125),
url VARCHAR(125),
checked BOOLEAN,
owner_by INT(10),
checked_date DATE
);
